Q6.Find the Winner

       
var Game_Winner = (s) => 
{
  
  let countA=0, countD=0;
  for(let i=0;i < s.length;i++){
       if(s[i]=="A"){
         countA++;
       }
       else{
         countD++
       }
  }
  
  if(countA>countD){
    return "Aditya";
  }
  else if(countD>countA){
    return "Danish";
  }
  else{
    return "Draw";
  }
};
